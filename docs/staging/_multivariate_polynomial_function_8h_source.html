62  SimTKMultivariatePolynomial(const SimTK::Vector_<T>& coefficients,
63  const int& dimension, const int& order) :
64  coefficients(coefficients), dimension(dimension), order(order) {
65  OPENSIM_THROW_IF(dimension < 0 || dimension > 4, Exception, format(
66  "Expected dimension >= 0 && <=4 but got %i.", dimension));
67  std::array<int, 4> nq {{0, 0, 0, 0}};
68  int coeff_nr = 0;
69  for (nq[0] = 0; nq[0] < order + 1; ++nq[0]) {
70  int nq2_s;
71  if (dimension < 2) nq2_s = 0;
72  else nq2_s = order - nq[0];
73  for (nq[1] = 0; nq[1] < nq2_s + 1; ++nq[1]) {
74  int nq3_s;
75  if (dimension < 3) nq3_s = 0;
76  else nq3_s = order - nq[0] - nq[1];
77  for (nq[2] = 0; nq[2] < nq3_s + 1; ++nq[2]) {
78  int nq4_s;
79  if (dimension < 4) nq4_s = 0;
80  else nq4_s = order - nq[0] - nq[1] - nq[2];
81  for (nq[3] = 0; nq[3] < nq4_s + 1; ++nq[3]) {
82  ++coeff_nr;
83  }
84  }
85  }
86  }
87  OPENSIM_THROW_IF(coefficients.size() != coeff_nr, Exception,
88  format("Expected %i coefficients but got %i.",
89  coeff_nr, coefficients.size()));

92  T calcValue(const SimTK::Vector& x) const override {
93  std::array<int, 4> nq {{0, 0, 0, 0}};
94  T value = static_cast<T>(0);
95  int coeff_nr = 0;
96  for (nq[0] = 0; nq[0] < order + 1; ++nq[0]) {
97  int nq2_s;
98  if (dimension < 2) nq2_s = 0;
99  else nq2_s = order - nq[0];
100  for (nq[1] = 0; nq[1] < nq2_s + 1; ++nq[1]) {
101  int nq3_s;
102  if (dimension < 3) nq3_s = 0;
103  else nq3_s = order - nq[0] - nq[1];
104  for (nq[2] = 0; nq[2] < nq3_s + 1; ++nq[2]) {
105  int nq4_s;
106  if (dimension < 4) nq4_s = 0;
107  else nq4_s = order - nq[0] - nq[1] - nq[2];
108  for (nq[3] = 0; nq[3] < nq4_s + 1; ++nq[3]) {
109  T valueP = static_cast<T>(1);
110  for (int i = 0; i < dimension; ++i) {
111  valueP *= std::pow(x[i], nq[i]);
112  }
113  value += valueP * coefficients[coeff_nr];
114  ++coeff_nr;
115  }
116  }
117  }
118  }

121  T calcDerivative(const SimTK::Array_<int>& derivComponent,
122  const SimTK::Vector& x) const override {
123  std::array<int, 4> nq {{0, 0, 0, 0}};
124  T value = static_cast<T>(0);
125  int nqNonNegative;
126  int coeff_nr = 0;
127  for (nq[0] = 0; nq[0] < order + 1; ++nq[0]) {
128  int nq2_s;
129  if (dimension < 2) nq2_s = 0;
130  else nq2_s = order - nq[0];
131  for (nq[1] = 0; nq[1] < nq2_s + 1; ++nq[1]) {
132  int nq3_s;
133  if (dimension < 3) nq3_s = 0;
134  else nq3_s = order - nq[0] - nq[1];
135  for (nq[2] = 0; nq[2] < nq3_s + 1; ++nq[2]) {
136  int nq4_s;
137  if (dimension < 4) nq4_s = 0;
138  else nq4_s = order - nq[0] - nq[1] - nq[2];
139  for (nq[3] = 0; nq[3] < nq4_s + 1; ++nq[3]) {
140  if (derivComponent[0] == 0) {
141  nqNonNegative = nq[0] - 1;
142  if (nqNonNegative < 0) nqNonNegative = 0;
143  T valueP = nq[0] * std::pow(x[0], nqNonNegative);
144  for (int i = 0; i < dimension; ++i) {
145  if (i == derivComponent[0]) continue;
146  valueP *= std::pow(x[i], nq[i]);
147  }
148  value += valueP * coefficients[coeff_nr];
149  }
150  else if (derivComponent[0] == 1) {
151  nqNonNegative = nq[1] - 1;
152  if (nqNonNegative < 0) nqNonNegative = 0;
153  T valueP = nq[1] * std::pow(x[1], nqNonNegative);
154  for (int i = 0; i < dimension; ++i) {
155  if (i == derivComponent[0]) continue;
156  valueP *= std::pow(x[i], nq[i]);
157  }
158  value += valueP * coefficients[coeff_nr];
159  }
160  else if (derivComponent[0] == 2) {
161  nqNonNegative = nq[2] - 1;
162  if (nqNonNegative < 0) nqNonNegative = 0;
163  T valueP = nq[2] * std::pow(x[2], nqNonNegative);
164  for (int i = 0; i < dimension; ++i) {
165  if (i == derivComponent[0]) continue;
166  valueP *= std::pow(x[i], nq[i]);
167  }
168  value += valueP * coefficients[coeff_nr];
169  }
170  else if (derivComponent[0] == 3) {
171  nqNonNegative = nq[3] - 1;
172  if (nqNonNegative < 0) nqNonNegative = 0;
173  T valueP = nq[3] * std::pow(x[3], nqNonNegative);
174  for (int i = 0; i < dimension; ++i) {
175  if (i == derivComponent[0]) continue;
176  valueP *= std::pow(x[i], nq[i]);
177  }
178  value += valueP * coefficients[coeff_nr];
179  }
180  ++coeff_nr;
181  }
182  }
183  }
184  }
185  return value;
