109  Model process(const std::string& relativeToDirectory = {}) const {
110  Model model;
111  if (get_filepath().empty()) {
112  if (!getProperty_model().empty()) {
113  model = get_model();
114  } else {
115  OPENSIM_THROW_FRMOBJ(Exception, "No source model.");
116  }
117  } else {
118  OPENSIM_THROW_IF_FRMOBJ(!getProperty_model().empty(), Exception,
119  "Expected either a Model object or a filepath, but "
120  "both were provided.");
121  std::string path = get_filepath();
122  if (!relativeToDirectory.empty()) {
123  using SimTK::Pathname;
124  path = Pathname::
125  getAbsolutePathnameUsingSpecifiedWorkingDirectory(
126  relativeToDirectory, path);
127  }
128  Model modelFromFile(path);
129  model = std::move(modelFromFile);
130  model.finalizeFromProperties();
131  model.finalizeConnections();
132  }
133 
134  for (int i = 0; i < getProperty_operators().size(); ++i) {
135  get_operators(i).operate(model, relativeToDirectory);
136  }
137  return model;
138  }
