198  CasOC::VariablesDM expandVariables(const casadi::DM& x) const {
199  CasOC::VariablesDM out;
200  using casadi::Slice;
201  casadi_int offset = 0;
202  for (const auto& key : getSortedVarKeys(m_vars)) {
203  const auto& value = m_vars.at(key);
204  // Convert a portion of the column vector into a matrix.
205  out[key] = casadi::DM::reshape(
206  x(Slice(offset, offset + value.numel())), value.rows(),
207  value.columns());
208  offset += value.numel();
209  }
210  return out;
211  }

216  template <typename T>
217  T flattenConstraints(const Constraints<T>& constraints) const {
218  T flat = T(casadi::Sparsity::dense(m_numConstraints, 1));
219 
220  int iflat = 0;
221  auto copyColumn = [&flat, &iflat](const T& matrix, int columnIndex) {
222  using casadi::Slice;
223  if (matrix.rows()) {
224  flat(Slice(iflat, iflat + matrix.rows())) =
225  matrix(Slice(), columnIndex);
226  iflat += matrix.rows();
227  }
228  };

230  // Trapezoidal sparsity pattern for mesh intervals 0, 1 and 2:
231  // Endpoint constraints depend on all time points through their
232  // integral.
233  // 0 1 2 3
234  // endpoint x x x x
235  // path_0 x
236  // kinematic_0 x
237  // residual_0 x
238  // defect_0 x x
239  // kinematic_1 x
240  // path_1 x
241  // residual_1 x
242  // defect_1 x x
243  // kinematic_2 x
244  // path_2 x
245  // residual_2 x
246  // kinematic_3 x
247  // path_3 x
248  // residual_3 x
249 
250  // Hermite-Simpson sparsity pattern for mesh intervals 0, 1 and 2:
251  // 0 0.5 1 1.5 2 2.5 3
252  // endpoint x x x x x x x
253  // path_0 x
254  // kinematic_0 x
255  // residual_0 x
256  // residual_0.5 x
257  // defect_0 x x x
258  // interp_con_0 x x x
259  // kinematic_1 x
260  // path_1 x
261  // residual_1 x
262  // residual_1.5 x
263  // defect_1 x x x
264  // interp_con_1 x x x
265  // kinematic_2 x
266  // path_2 x
267  // residual_2 x
268  // residual_2.5 x
269  // defect_2 x x x
270  // interp_con_2 x x x
271  // kinematic_3 x
272  // path_3 x
273  // residual_3 x
274  // 0 0.5 1 1.5 2 2.5 3
275 
276  for (const auto& endpoint : constraints.endpoint) {
277  copyColumn(endpoint, 0);
278  }
279 
280  int igrid = 0;
281  // Index for pointsForInterpControls.
282  int icon = 0;
283  for (int imesh = 0; imesh < m_numMeshPoints; ++imesh) {
284  copyColumn(constraints.kinematic, imesh);
285  for (const auto& path : constraints.path) {
286  copyColumn(path, imesh);
287  }
288  if (imesh < m_numMeshIntervals) {
289  while (m_grid(igrid).scalar() < m_solver.getMesh()[imesh + 1]) {
290  copyColumn(constraints.multibody_residuals, igrid);
291  copyColumn(constraints.auxiliary_residuals, igrid);
292  ++igrid;
293  }
294  copyColumn(constraints.defects, imesh);
295  while (icon < m_pointsForInterpControls.numel() &&
296  m_pointsForInterpControls(icon).scalar() <
297  m_solver.getMesh()[imesh + 1]) {
298  copyColumn(constraints.interp_controls, icon);
299  ++icon;
300  }
301  }
302  }
303  // The loop above does not handle the residual at the final grid point.
304  copyColumn(constraints.multibody_residuals, m_numGridPoints - 1);
305  copyColumn(constraints.auxiliary_residuals, m_numGridPoints - 1);
306 
307  OPENSIM_THROW_IF(iflat != m_numConstraints, OpenSim::Exception,
308  "Internal error.");
309  return flat;
310  }

312  // Expand constraints that have been flattened into a Constraints struct.
313  template <typename T>
314  Constraints<T> expandConstraints(const T& flat) const {
315  using casadi::Sparsity;
316 
317  // Allocate memory.
318  auto init = [](int numRows, int numColumns) {
319  return T(casadi::Sparsity::dense(numRows, numColumns));
320  };
321  Constraints<T> out;
322  out.defects = init(m_numDefectsPerMeshInterval, m_numMeshPoints - 1);
323  out.multibody_residuals = init(m_numMultibodyResiduals,
324  m_numGridPoints);
325  out.auxiliary_residuals = init(m_numAuxiliaryResiduals,
326  m_numGridPoints);
327  out.kinematic = init(m_problem.getNumKinematicConstraintEquations(),
328  m_numMeshPoints);
329  out.endpoint.resize(m_problem.getEndpointConstraintInfos().size());
330  for (int iec = 0; iec < (int)m_constraints.endpoint.size(); ++iec) {
331  const auto& info = m_problem.getEndpointConstraintInfos()[iec];
332  out.endpoint[iec] = init(info.num_outputs, 1);
333  }
334  out.path.resize(m_problem.getPathConstraintInfos().size());
335  for (int ipc = 0; ipc < (int)m_constraints.path.size(); ++ipc) {
336  const auto& info = m_problem.getPathConstraintInfos()[ipc];
337  out.path[ipc] = init(info.size(), m_numMeshPoints);
338  }
339  out.interp_controls = init(m_problem.getNumControls(),
340  (int)m_pointsForInterpControls.numel());
341 
342  int iflat = 0;
343  auto copyColumn = [&flat, &iflat](T& matrix, int columnIndex) {
344  using casadi::Slice;
345  if (matrix.rows()) {
346  matrix(Slice(), columnIndex) =
347  flat(Slice(iflat, iflat + matrix.rows()));
348  iflat += matrix.rows();
349  }
350  };
351 
352  for (auto& endpoint : out.endpoint) {
353  copyColumn(endpoint, 0);
354  }
355 
356  int igrid = 0;
357  // Index for pointsForInterpControls.
358  int icon = 0;
359  for (int imesh = 0; imesh < m_numMeshPoints; ++imesh) {
360  copyColumn(out.kinematic, imesh);
361  for (auto& path : out.path) { copyColumn(path, imesh); }
362  if (imesh < m_numMeshIntervals) {
363  while (m_grid(igrid).scalar() < m_solver.getMesh()[imesh + 1]) {
364  copyColumn(out.multibody_residuals, igrid);
365  copyColumn(out.auxiliary_residuals, igrid);
366  ++igrid;
367  }
368  copyColumn(out.defects, imesh);
369  while (icon < m_pointsForInterpControls.numel() &&
370  m_pointsForInterpControls(icon).scalar() <
371  m_solver.getMesh()[imesh + 1]) {
372  copyColumn(out.interp_controls, icon);
373  ++icon;
374  }
375  }
376  }
377  // The loop above does not handle residuals at the final grid point.
378  copyColumn(out.multibody_residuals, m_numGridPoints - 1);
379  copyColumn(out.auxiliary_residuals, m_numGridPoints - 1);
380 
381  OPENSIM_THROW_IF(iflat != m_numConstraints, OpenSim::Exception,
382  "Internal error.");
383  return out;
384  }
