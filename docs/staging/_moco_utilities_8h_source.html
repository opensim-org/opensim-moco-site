190 template <typename TimeVector, typename FunctionType = GCVSpline>
191 TimeSeriesTable resample(const TimeSeriesTable& in, const TimeVector& newTime) {
192 
193  const auto& time = in.getIndependentColumn();
194 
195  OPENSIM_THROW_IF(time.size() < 2, Exception,
196  "Cannot resample if number of times is 0 or 1.");
197  OPENSIM_THROW_IF(newTime[0] < time[0], Exception,
198  format("New initial time (%f) cannot be less than existing "
199  "initial time (%f)",
200  newTime[0], time[0]));
201  OPENSIM_THROW_IF(newTime[newTime.size() - 1] > time[time.size() - 1],
202  Exception,
203  format("New final time (%f) cannot be greater than existing final "
204  "time (%f)",
205  newTime[newTime.size() - 1], time[time.size() - 1]));
206  for (int itime = 1; itime < (int)newTime.size(); ++itime) {
207  OPENSIM_THROW_IF(newTime[itime] < newTime[itime - 1], Exception,
208  format("New times must be non-decreasing, but "
209  "time[%i] < time[%i] (%f < %f).",
210  itime, itime - 1, newTime[itime], newTime[itime - 1]));
211  }
212 
213  // Copy over metadata.
214  TimeSeriesTable out = in;
215  for (int irow = (int)out.getNumRows() - 1; irow >= 0; --irow) {
216  out.removeRowAtIndex(irow);
217  }
218 
219  std::unique_ptr<FunctionSet> functions =
220  createFunctionSet<FunctionType>(in);
221  SimTK::Vector curTime(1);
222  SimTK::RowVector row(functions->getSize());
223  for (int itime = 0; itime < (int)newTime.size(); ++itime) {
224  curTime[0] = newTime[itime];
225  for (int icol = 0; icol < functions->getSize(); ++icol) {
226  row(icol) = functions->get(icol).calcValue(curTime);
227  }
228  // Not efficient!
229  out.appendRow(curTime[0], row);
230  }
231  return out;
232 }

305 template <typename T>
306 TimeSeriesTable_<T> analyze(Model model, const MocoTrajectory& iterate,
307  std::vector<std::string> outputPaths) {
308 
309  // Initialize the system so we can access the outputs.
310  model.initSystem();
311  // Create the reporter object to which we'll add the output data to create
312  // the report.
313  auto* reporter = new TableReporter_<T>();
314  // Loop through all the outputs for all components in the model, and if
315  // the output path matches one provided in the argument and the output type
316  // agrees with the template argument type, add it to the report.
317  for (const auto& comp : model.getComponentList()) {
318  for (const auto& outputName : comp.getOutputNames()) {
319  const auto& output = comp.getOutput(outputName);
320  auto thisOutputPath = output.getPathName();
321  for (const auto& outputPathArg : outputPaths) {
322  if (std::regex_match(
323  thisOutputPath, std::regex(outputPathArg))) {
324  // Make sure the output type agrees with the template.
325  if (dynamic_cast<const Output<T>*>(&output)) {
326  reporter->addToReport(output);
327  } else {
328  std::cout << format("Warning: ignoring output %s of "
329  "type %s.",
330  output.getPathName(),
331  output.getTypeName())
332  << std::endl;
333  }
334  }
335  }
336  }
337  }
338  model.addComponent(reporter);
339  model.initSystem();
340 
341  // Get states trajectory.
342  Storage storage = iterate.exportToStatesStorage();
343  auto statesTraj = StatesTrajectory::createFromStatesStorage(model, storage);
344 
345  // Loop through the states trajectory to create the report.
346  for (int i = 0; i < (int)statesTraj.getSize(); ++i) {
347  // Get the current state.
348  auto state = statesTraj[i];
349 
350  // Enforce any SimTK::Motion's included in the model.
351  model.getSystem().prescribe(state);
352 
353  // Create a SimTK::Vector of the control values for the current state.
354  SimTK::RowVector controlsRow = iterate.getControlsTrajectory().row(i);
355  SimTK::Vector controls(controlsRow.size(),
356  controlsRow.getContiguousScalarData(), true);
357 
358  // Set the controls on the state object.
359  model.realizeVelocity(state);
360  model.setControls(state, controls);
361 
362  // Generate report results for the current state.
363  model.realizeReport(state);
364  }
365 
366  return reporter->getTable();
367 }

443 template <typename T = double>
444 std::vector<SimTK::ReferencePtr<const Output<T>>> getModelOutputReferencePtrs(
445  const Component& component, const std::string& pattern,
446  bool includeDescendents = false) {
447 
448  // Create regex.
449  std::regex regex(pattern);
450  // Initialize outputs array.
451  std::vector<SimTK::ReferencePtr<const Output<T>>> outputs;
452 
453  std::function<void(const Component&, const std::regex&, bool,
454  std::vector<SimTK::ReferencePtr<const Output<T>>>&)> helper;
455  helper = [&helper](const Component& component, const std::regex& regex,
456  bool includeDescendents,
457  std::vector<SimTK::ReferencePtr<const Output<T>>>& outputs) {
458  // Store a reference to outputs that match the template
459  // parameter type and whose names contain the provided
460  // substring.
461  for (const auto& entry : component.getOutputs()) {
462  const std::string& name = entry.first;
463  const auto foundSubstring = std::regex_match(name, regex);
464  const auto* output =
465  dynamic_cast<const Output<T>*>(entry.second.get());
466  if (output && foundSubstring) {
467  outputs.emplace_back(output);
468  }
469  }
470 
471  // Repeat for all subcomponents.
472  if (includeDescendents) {
473  for (const Component& thisComp :
474  component.getComponentList<Component>()) {
475  if (&thisComp == &component) { continue; }
476  helper(thisComp, regex, false, outputs);
477  }
478  }
479  };
480 
481  helper(component, regex, includeDescendents, outputs);
482  return outputs;
483 }

624  static std::string formatNs(const long long& nanoseconds) {
625  std::stringstream ss;
626  double seconds = SimTK::nsToSec(nanoseconds);
627  int secRounded = (int)std::round(seconds);
628  if (seconds > 1)
629  ss << secRounded << " second(s)";
630  else if (nanoseconds >= 1000000)
631  ss << nanoseconds / 1000000 << " millisecond(s)";
632  else if (nanoseconds >= 1000)
633  ss << nanoseconds / 1000 << " microsecond(s)";
634  else
635  ss << nanoseconds << " nanosecond(s)";
636  int minutes = secRounded / 60;
637  int hours = minutes / 60;
638  if (minutes || hours) {
639  ss << " (";
640  if (hours) {
641  ss << hours << " hour(s), ";
642  ss << minutes % 60 << " minute(s), ";
643  ss << secRounded % 60 << " second(s)";
644  } else {
645  ss << minutes % 60 << " minute(s), ";
646  ss << secRounded % 60 << " second(s)";
647  }
648  ss << ")";
649  }
650  return ss.str();
651  }

673 template <typename T> class ThreadsafeJar {
674 public:
678  std::unique_ptr<T> take() {
679  // Only one thread can lock the mutex at a time, so only one thread
680  // at a time can be in any of the functions of this class.
681  std::unique_lock<std::mutex> lock(m_mutex);
682  // Block this thread until the condition variable is woken up
683  // (by a notify_...()) and the lambda function returns true.
684  m_inventoryMonitor.wait(lock, [this] { return m_entries.size() > 0; });
685  std::unique_ptr<T> top = std::move(m_entries.top());
686  m_entries.pop();
687  return top;
688  }
692  void leave(std::unique_ptr<T> entry) {
693  std::unique_lock<std::mutex> lock(m_mutex);
694  m_entries.push(std::move(entry));
695  lock.unlock();
696  m_inventoryMonitor.notify_one();
697  }
699  int size() const {
700  std::lock_guard<std::mutex> lock(m_mutex);
701  return (int)m_entries.size();
702  }
703 
704 private:
705  std::stack<std::unique_ptr<T>> m_entries;
706  mutable std::mutex m_mutex;
707  std::condition_variable m_inventoryMonitor;
708 };
