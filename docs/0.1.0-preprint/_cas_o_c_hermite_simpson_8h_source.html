43 class HermiteSimpson : public Transcription {
44 public:
45  HermiteSimpson(const Solver& solver, const Problem& problem)
46  : Transcription(solver, problem) {
47  casadi::DM grid =
48  casadi::DM::zeros(1, (2 * m_solver.getMesh().size()) - 1);
49  const auto& mesh = m_solver.getMesh();
50  const bool interpControls = m_solver.getInterpolateControlMidpoints();
51  casadi::DM pointsForInterpControls;
52  if (interpControls) {
53  pointsForInterpControls =
54  casadi::DM::zeros(1, m_solver.getMesh().size() - 1);
55  }
56  for (int i = 0; i < grid.numel(); ++i) {
57  if (i % 2 == 0) {
58  grid(i) = mesh[i / 2];
59  } else {
60  grid(i) = .5 * (mesh[i / 2] + mesh[i / 2 + 1]);
61  if (interpControls) {
62  pointsForInterpControls(i / 2) = grid(i);
63  }
64  }
65  }
66  createVariablesAndSetBounds(grid, 2 * m_problem.getNumStates(),
67  pointsForInterpControls);
68  }
