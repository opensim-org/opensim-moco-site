251  SimTK::Vector calcKinematicConstraintErrors(
252  const SimTK::State& state) const {
253  SimTK::Vector errors(getNumKinematicConstraintEquations(), 0.0);
254  int index = 0;
255  int thisConstraintNumEquations;
256  for (int i = 0; i < (int)m_kinematic_constraints.size(); ++i) {
257  thisConstraintNumEquations = m_kinematic_constraints[i]
258  .getConstraintInfo()
259  .getNumEquations();
260 
261  SimTK::Vector theseErrors(thisConstraintNumEquations,
262  errors.getContiguousScalarData() + index, true);
263  m_kinematic_constraints[i].calcKinematicConstraintErrors(
264  getModelBase(), state, theseErrors);
265 
266  index += thisConstraintNumEquations;
267  }
268  return errors;
269  }
