226 template <typename TimeVector, typename FunctionType = GCVSpline>
227 TimeSeriesTable resample(const TimeSeriesTable& in, const TimeVector& newTime) {
228 
229  const auto& time = in.getIndependentColumn();
230 
231  OPENSIM_THROW_IF(time.size() < 2, Exception,
232  "Cannot resample if number of times is 0 or 1.");
233  OPENSIM_THROW_IF(newTime[0] < time[0], Exception,
234  format("New initial time (%f) cannot be less than existing "
235  "initial time (%f)",
236  newTime[0], time[0]));
237  OPENSIM_THROW_IF(newTime[newTime.size() - 1] > time[time.size() - 1],
238  Exception,
239  format("New final time (%f) cannot be greater than existing final "
240  "time (%f)",
241  newTime[newTime.size() - 1], time[time.size() - 1]));
242  for (int itime = 1; itime < (int)newTime.size(); ++itime) {
243  OPENSIM_THROW_IF(newTime[itime] < newTime[itime - 1], Exception,
244  format("New times must be non-decreasing, but "
245  "time[%i] < time[%i] (%f < %f).",
246  itime, itime - 1, newTime[itime], newTime[itime - 1]));
247  }
248 
249  // Copy over metadata.
250  TimeSeriesTable out = in;
251  for (int irow = (int)out.getNumRows() - 1; irow >= 0; --irow) {
252  out.removeRowAtIndex(irow);
253  }
254 
255  std::unique_ptr<FunctionSet> functions =
256  createFunctionSet<FunctionType>(in);
257  SimTK::Vector curTime(1);
258  SimTK::RowVector row(functions->getSize());
259  for (int itime = 0; itime < (int)newTime.size(); ++itime) {
260  curTime[0] = newTime[itime];
261  for (int icol = 0; icol < functions->getSize(); ++icol) {
262  row(icol) = functions->get(icol).calcValue(curTime);
263  }
264  // Not efficient!
265  out.appendRow(curTime[0], row);
266  }
267  return out;
268 }

329 template <typename T>
330 TimeSeriesTable_<T> analyze(Model model, const MocoTrajectory& trajectory,
331  std::vector<std::string> outputPaths) {
332 
333  // Initialize the system so we can access the outputs.
334  model.initSystem();
335  // Create the reporter object to which we'll add the output data to create
336  // the report.
337  auto* reporter = new TableReporter_<T>();
338  // Loop through all the outputs for all components in the model, and if
339  // the output path matches one provided in the argument and the output type
340  // agrees with the template argument type, add it to the report.
341  for (const auto& comp : model.getComponentList()) {
342  for (const auto& outputName : comp.getOutputNames()) {
343  const auto& output = comp.getOutput(outputName);
344  auto thisOutputPath = output.getPathName();
345  for (const auto& outputPathArg : outputPaths) {
346  if (std::regex_match(
347  thisOutputPath, std::regex(outputPathArg))) {
348  // Make sure the output type agrees with the template.
349  if (dynamic_cast<const Output<T>*>(&output)) {
350  reporter->addToReport(output);
351  } else {
352  std::cout << format("Warning: ignoring output %s of "
353  "type %s.",
354  output.getPathName(),
355  output.getTypeName())
356  << std::endl;
357  }
358  }
359  }
360  }
361  }
362  model.addComponent(reporter);
363  model.initSystem();
364 
365  // Get states trajectory.
366  Storage storage = trajectory.exportToStatesStorage();
367  auto statesTraj = StatesTrajectory::createFromStatesStorage(model, storage);
368 
369  // Loop through the states trajectory to create the report.
370  for (int i = 0; i < (int)statesTraj.getSize(); ++i) {
371  // Get the current state.
372  auto state = statesTraj[i];
373 
374  // Enforce any SimTK::Motion's included in the model.
375  model.getSystem().prescribe(state);
376 
377  // Create a SimTK::Vector of the control values for the current state.
378  SimTK::RowVector controlsRow =
379  trajectory.getControlsTrajectory().row(i);
380  SimTK::Vector controls(controlsRow.size(),
381  controlsRow.getContiguousScalarData(), true);
382 
383  // Set the controls on the state object.
384  model.realizeVelocity(state);
385  model.setControls(state, controls);
386 
387  // Generate report results for the current state.
388  model.realizeReport(state);
389  }
390 
391  return reporter->getTable();
392 }

484 template <typename T = double>
485 std::vector<SimTK::ReferencePtr<const Output<T>>> getModelOutputReferencePtrs(
486  const Component& component, const std::string& pattern,
487  bool includeDescendents = false) {
488 
489  // Create regex.
490  std::regex regex(pattern);
491  // Initialize outputs array.
492  std::vector<SimTK::ReferencePtr<const Output<T>>> outputs;
493 
494  std::function<void(const Component&, const std::regex&, bool,
495  std::vector<SimTK::ReferencePtr<const Output<T>>>&)> helper;
496  helper = [&helper](const Component& component, const std::regex& regex,
497  bool includeDescendents,
498  std::vector<SimTK::ReferencePtr<const Output<T>>>& outputs) {
499  // Store a reference to outputs that match the template
500  // parameter type and whose names contain the provided
501  // substring.
502  for (const auto& entry : component.getOutputs()) {
503  const std::string& name = entry.first;
504  const auto foundSubstring = std::regex_match(name, regex);
505  const auto* output =
506  dynamic_cast<const Output<T>*>(entry.second.get());
507  if (output && foundSubstring) {
508  outputs.emplace_back(output);
509  }
510  }
511 
512  // Repeat for all subcomponents.
513  if (includeDescendents) {
514  for (const Component& thisComp :
515  component.getComponentList<Component>()) {
516  if (&thisComp == &component) { continue; }
517  helper(thisComp, regex, false, outputs);
518  }
519  }
520  };
521 
522  helper(component, regex, includeDescendents, outputs);
523  return outputs;
524 }

686  static std::string formatNs(const long long& nanoseconds) {
687  std::stringstream ss;
688  double seconds = SimTK::nsToSec(nanoseconds);
689  int secRounded = (int)std::round(seconds);
690  if (seconds > 1)
691  ss << secRounded << " second(s)";
692  else if (nanoseconds >= 1000000)
693  ss << nanoseconds / 1000000 << " millisecond(s)";
694  else if (nanoseconds >= 1000)
695  ss << nanoseconds / 1000 << " microsecond(s)";
696  else
697  ss << nanoseconds << " nanosecond(s)";
698  int minutes = secRounded / 60;
699  int hours = minutes / 60;
700  if (minutes || hours) {
701  ss << " (";
702  if (hours) {
703  ss << hours << " hour(s), ";
704  ss << minutes % 60 << " minute(s), ";
705  ss << secRounded % 60 << " second(s)";
706  } else {
707  ss << minutes % 60 << " minute(s), ";
708  ss << secRounded % 60 << " second(s)";
709  }
710  ss << ")";
711  }
712  return ss.str();
713  }

716  long long m_startTime;

736 // TODO: Find a way to always give the same thread the same object.
737 template <typename T> class ThreadsafeJar {
738 public:
742  std::unique_ptr<T> take() {
743  // Only one thread can lock the mutex at a time, so only one thread
744  // at a time can be in any of the functions of this class.
745  std::unique_lock<std::mutex> lock(m_mutex);
746  // Block this thread until the condition variable is woken up
747  // (by a notify_...()) and the lambda function returns true.
748  m_inventoryMonitor.wait(lock, [this] { return m_entries.size() > 0; });
749  std::unique_ptr<T> top = std::move(m_entries.top());
750  m_entries.pop();
751  return top;
752  }
756  void leave(std::unique_ptr<T> entry) {
757  std::unique_lock<std::mutex> lock(m_mutex);
758  m_entries.push(std::move(entry));
759  lock.unlock();
760  m_inventoryMonitor.notify_one();
761  }
763  int size() const {
764  std::lock_guard<std::mutex> lock(m_mutex);
765  return (int)m_entries.size();
766  }
767 
768 private:
769  std::stack<std::unique_ptr<T>> m_entries;
770  mutable std::mutex m_mutex;
771  std::condition_variable m_inventoryMonitor;
772 };
