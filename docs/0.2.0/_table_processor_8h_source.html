80  TimeSeriesTable process(std::string relativeToDirectory,
81  const Model* model = nullptr) const {
82  TimeSeriesTable table;
83  if (get_filepath().empty()) {
84  if (m_tableProvided) {
85  table = m_table;
86  } else {
87  OPENSIM_THROW_FRMOBJ(Exception, "No source table.");
88  }
89  } else {
90  OPENSIM_THROW_IF_FRMOBJ(m_tableProvided, Exception,
91  "Expected either an in-memory table or a filepath, but "
92  "both were provided.");
93  std::string path = get_filepath();
94  if (!relativeToDirectory.empty()) {
95  using SimTK::Pathname;
96  path = Pathname::
97  getAbsolutePathnameUsingSpecifiedWorkingDirectory(
98  relativeToDirectory, path);
99  }
100  table = TimeSeriesTable(path);
101  }
102 
103  if (table.hasTableMetaDataKey("inDegrees") &&
104  table.getTableMetaDataAsString("inDegrees") == "yes") {
105  model->getSimbodyEngine().convertDegreesToRadians(table);
106  }
107 
108  for (int i = 0; i < getProperty_operators().size(); ++i) {
109  get_operators(i).operate(table, model);
110  }
111  return table;
112  }
115  TimeSeriesTable process(const Model* model = nullptr) const {
116  return process({}, model);
117  }
120  bool empty() const {
121  return !m_tableProvided && get_filepath().empty();
122  }
124  TableProcessor& append(const TableOperator& op) {
125  append_operators(op);
126  return *this;
127  }
130  TableProcessor& append(const TableProcessor& traj) {
131  for (int i = 0; i < traj.getProperty_operators().size(); ++i) {
132  append_operators(traj.get_operators(i));
133  }
134  return *this;
135  }
141  TableProcessor& operator|(const TableOperator& right) {
142  return append(right);
143  }
144 
145 private:
146  bool m_tableProvided = false;
147  TimeSeriesTable m_table;
148 };
